function isPalindrome(num) { const str = num.toString(); return str === str.split('').reverse().join(''); } function nextPalindrome(n) { let next = n + 1; while (!isPalindrome(next)) { next++; } return next; } // Input via prompt, Output via alert const input = parseInt(prompt("Enter a number:")); const result = nextPalindrome(input); alert(result);
